with cte1 as (
select 
	*
	, unnest(string_to_array(filter_room_types, ',')) as room_type
from airbnb_searches),
cte2 as (
select
	room_type
	, count(room_type)
from cte1
group by room_type)

select
	room_type
	, count
from (
select 
	*
	, rank() over(order by count desc)
from cte2) a
where rank = 1;
