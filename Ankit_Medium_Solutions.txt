with cte1 as (
select 
	*
	, unnest(string_to_array(filter_room_types, ',')) as room_type
from airbnb_searches),
cte2 as (
select
	room_type
	, count(room_type)
from cte1
group by room_type)

select
	room_type
	, count
from (
select 
	*
	, rank() over(order by count desc)
from cte2) a
where rank = 1;

with cte1 as (
select
  a.artist_name
  ,count(s.name) as cnt
from artists a
inner JOIN songs s
on a.artist_id=s.artist_id
inner join (select * from global_song_rank where rank<11) gs
on s.song_id=gs.song_id
group by a.artist_name)

select * from (
select 
  artist_name
  ,dense_rank() over(order by cnt desc)
from cte1) a
where dense_rank<6;
